clc
clear all
close all
warning off
load Classifier

[filename, pathname] = uigetfile('*.*', 'Pick an Image');
filename = strcat(pathname, filename);

TestImage = imread(filename);
ScaledTestImage = imresize(TestImage, [256, 256]);
TestFeatures =  extractHOGFeatures(ScaledTestImage);
PredictIndex = predict(classifier, TestFeatures);
figure;
imshow(ScaledTestImage);
title(['Image predicted as: ', char(PredictIndex)]);
